        // --- PTI LOGIC ---
        function addDirective(problem, text, role) {
            if (!currentPTI[problem]) currentPTI[problem] = [];
            if (currentPTI[problem].some(d => d.text === text)) return;
            currentPTI[problem].push({ text, role });
            updateSidebar();
            
            const badge = document.getElementById('pti-badge');
            badge.classList.remove('hidden');
            badge.classList.add('animate-bounce');
            setTimeout(() => badge.classList.remove('animate-bounce'), 1000);
        }

        function removeDirective(problem, index) {
            currentPTI[problem].splice(index, 1);
            if (currentPTI[problem].length === 0) delete currentPTI[problem];
            updateSidebar();
        }

        function updateSidebar() {
            const tbody = document.getElementById('pti-table-body');
            const visualTable = document.getElementById('pti-visual-table');
            const emptyState = document.getElementById('empty-state');
            const badge = document.getElementById('pti-badge');

            tbody.innerHTML = '';
            const problems = Object.keys(currentPTI);
            
            let totalDirs = 0;
            problems.forEach(p => totalDirs += currentPTI[p].length);
            badge.innerText = totalDirs;
            badge.classList.toggle('hidden', totalDirs === 0);

            if (problems.length === 0) {
                visualTable.classList.add('hidden');
                emptyState.classList.remove('hidden');
                return;
            }

            visualTable.classList.remove('hidden');
            emptyState.classList.add('hidden');

            problems.forEach(prob => {
                const directives = currentPTI[prob];
                const tr = document.createElement('tr');
                const dirHtml = directives.map((d, i) => `
                    <div class="flex justify-between items-start mb-1 group hover:bg-slate-50 p-1 rounded border-b border-dashed border-slate-100 last:border-0">
                        <div class="text-xs">
                            <span class="role-badge role-${d.role.toLowerCase()} text-[9px] mr-1 scale-90 origin-left">[${d.role}]</span> 
                            <span class="text-slate-700">${d.text}</span>
                        </div>
                        <button onclick="removeDirective('${prob}', ${i})" class="text-slate-300 hover:text-red-500 ml-2 no-print p-1">
                            <i class="fa-solid fa-trash-can"></i>
                        </button>
                    </div>
                `).join('');
                tr.innerHTML = `<td class="font-bold text-slate-700 bg-slate-50 text-xs align-top border-r border-slate-200">${prob}</td><td class="align-top">${dirHtml}</td>`;
                tbody.appendChild(tr);
            });
        }

        // --- AI LOGIC (CALLS) ---
        async function callGemini(prompt, system) {
            const key = getApiKey();
            if (!key) return "Clé API manquante.";
            const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${key}`;
            const payload = { contents: [{ parts: [{ text: prompt }] }], systemInstruction: { parts: [{ text: system }] } };
            try {
                const res = await fetch(url, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(payload) });
                if(!res.ok) throw new Error("API Error");
                const data = await res.json();
                return data.candidates?.[0]?.content?.parts?.[0]?.text || "Erreur de réponse.";
            } catch (e) {
                return "Erreur de connexion avec l'IA. Vérifiez votre clé ou connexion.";
            }
        }

            currentTab = mode;

            const manualInput = document.getElementById('manual-input-section');

            btnCoach.className = "flex-1 py-3 text-sm font-medium transition-colors inactive-tab";
            btnManual.className = "flex-1 py-3 text-sm font-medium transition-colors inactive-tab";

            if (mode === 'coach') {

                manualInput.classList.add('hidden');

            } else if (mode === 'manual') {
                btnManual.className = "flex-1 py-3 text-sm font-medium transition-colors active-tab";
                chatInput.classList.add('hidden');
                manualInput.classList.remove('hidden'); // Show Manual Input
